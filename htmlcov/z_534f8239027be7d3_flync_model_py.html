107 @model_validator(mode="before") 

108 @classmethod 

109 def skip_broken_ecus(cls, data): 

110 """ 

111 Remove None ECUs from the list before validation. 

112 

113 When an ECU file fails to load the workspace inserts None into the ecus list. 

114 JErrors are already reported at the ECU level, so the None entries are silently dropped here to prevent a cascade of 

115 FLYNCModel-level errors for the same root cause. 

116 """ 

117 

118 if isinstance(data, dict): 

119 ecus = data.get("ecus") or [] 

120 if isinstance(ecus, list) and any(e is None for e in ecus): 

121 data["ecus"] = [e for e in ecus if e is not None] 

122 return data

124 def model_post_init(self, context): 

125 """ 

126 Perform post-initialization processing after the model is created. 

127 

128 Following steps are performed: 

129 

130 1. Populate the solicited-node RX multicast group memberships for each IPv6 address configured in any ECU. 

131 

132 2. Populate the solicited-node TX multicast group memberships for each ECU based on the RX entries for the same multicast group and VLAN. 

133 """ 

134 

135 self.__populate_ipv6_solicited_node_multicasts_rx() 

136 self.__populate_ipv6_solicited_node_multicasts_tx()

138 @model_validator(mode="after") 

139 def validate_unique_ips(self): 

140 """ 

141 Validate all IPs are unique system wide 

142 """ 

143 

144 try: 

145 all_ips = [] 

146 for ecu in self.ecus: 

147 new_ips = ecu.get_all_ips() 

148 for ip in new_ips: 

149 if ip not in all_ips: 

150 all_ips.append(ip) 

151 elif str(ip) not in ("0.0.0.0", "::"): 

152 warn(f"The IP {ip} is repeated in ECU {ecu.name}") 

153 except PydanticCustomError as e: 

154 warn(str(e)) 

155 return self

157 @model_validator(mode="after") 

158 def check_tx_rx_multicast_group(self): 

159 try: 

160 tx_list = [] 

161 rx_list = [] 

162 separ = "/VLAN" 

163 for ecu in self.ecus: 

164 for mcast in ecu.multicast_groups: 

165 key = str(mcast.group) + separ + str(mcast.vlan) 

166 if mcast.mode == "tx": 

167 tx_list.append(key) 

168 if mcast.mode == "rx": 

169 rx_list.append(key) 

170 

171 for rx in rx_list: 

172 if rx not in tx_list: 

173 warn(f"Invalid Multicast Configuration. There is a multicast rx configured for the address {rx} but no tx.") 

174 except PydanticCustomError as e: 

175 warn(str(e)) 

176 return self 

177 

178 @model_validator(mode="after") 

179 def validate_multicast_paths(self): 

180 try: 

181 paths = dict() 

182 vlans_dict = dict() 

183 separ = "/VLAN" 

184 for ecu in self.ecus: 

185 for mcast in ecu.multicast_groups: 

186 key = str(mcast.group) + separ + str(mcast.vlan) 

187 vlans_dict[key] = mcast.vlan 

188 if (mcast.mode == "tx") and key not in paths: 

189 

190 paths[key] = compute_path(mcast.vlan, mcast._interface) 

191 if (mcast.mode == "tx") and key in paths and not check_obj_in_list(mcast._interface, paths[key]): 

192 warn( 

193 "Invalid Multicast Address Configuration. There are several RX that the TX Endpoint at " 

194 f"{mcast._interface.name} cannot reach. {serialize_components(paths[key])}" 

195 ) 

196 self.check_rx_are_reached(separ, paths, vlans_dict) 

197 except PydanticCustomError as e: 

198 warn(str(e)) 

199 return self

201 @model_validator(mode="after") 

202 def validate_multicast_someip(self): 

203 """ 

204 Validate multicast configuration for SOME/IP consumers and providers 

205 

206 For provider: check if the parent socket has a multicast_tx entry 

207 """ 

208 

209 deployments = [ 

210 (deployment.root, socket, ecu) 

211 for ecu in self.ecus 

212 for ctrl in ecu.controllers 

213 for iface in ctrl.ethernet_interfaces 

214 for sock_con in iface.sockets 

215 for socket in sock_con.sockets 

216 for deployment in socket.deployments 

217 if deployment.root.deployment_type.startswith("someip_") and socket.endpoint_type == "multicast" and socket.protocol == "udp" 

218 ] 

219 

220 providers = [dpl for dpl in deployments if dpl[0].deployment_type == "someip_provider"] 

221 

222 # Providers need to have multicast_tx in socket 

223 for provider, socket, _ecu in providers: 

224 for mcast_config in provider.multicast_config or []: 

225 if mcast_config.ip_address not in socket.multicast_tx: 

226 raise err_major( 

227 f"Deployed provided service ({provider.service.name}, {provider.service.id:#06x}, {provider.service.major_version}) " 

228 f"has multicast configuration for eventgroups ({mcast_config.eventgroups}/{mcast_config.ip_address}), " 

229 f"but socket ({socket.name}) does not indicate by multicast_tx entry ({socket.multicast_tx})" 

230 ) 

231 

232 return self

259 def check_rx_are_reached(self, separ, paths, vlans_dict): 

260 for ecu in self.ecus: 

261 for mcast in ecu.multicast_groups: 

262 key = str(mcast.group) + separ + str(mcast.vlan) 

263 if (mcast.mode == "rx") and key not in paths: 

264 

265 warn("Invalid Multicast Address Configuration. There are no TX endpoints for this address {key} ") 

266 if (mcast.mode == "rx") and key in paths and not check_obj_in_list(mcast._interface, paths[key]): 

267 warn( 

268 f"Invalid Multicast Address Configuration. The RX interface for address {key} " 

269 f"- {mcast._interface.name} cannot be reached by the TX ports." 

270 ) 

271 

272 self.load_switch_multicast(vlans_dict, paths) 

273 

274 return self

276 def __populate_ipv6_solicited_node_multicasts_rx(self): 

277 """ 

278 Populate the solicited-node multicast group memberships for each IPv6 address configured in any ECU. 

279 """ 

280 

281 for ecu in self.ecus: 

282 update_ecu_multicast = collect_ipv6_solicited_node_rx(ecu) 

283 if ecu.name in update_ecu_multicast: 

284 ecu.multicast_groups.append(update_ecu_multicast[ecu.name]) 

285 return self

287 def __populate_ipv6_solicited_node_multicasts_tx(self): 

288 """ 

289 Populate the solicited-node multicast group memberships for each IPv6 address configured in any ECU as TX if there is a RX for the 

290 same multicast group and VLAN. 

291 """ 

292 

293 multicasts = [mc for ecu in self.ecus for mc in ecu.multicast_groups if mc.solicited_node_multicast] 

294 

295 for ecu in self.ecus: 

296 update_ecu_multicast = collect_ipv6_solicited_node_tx(ecu, multicasts) 

297 if ecu.name in update_ecu_multicast: 

298 ecu.multicast_groups.append(update_ecu_multicast[ecu.name]) 

299 return self

301 def append_mcast(self, vlan, comp, mcast_addr): 

302 for v_entry in comp.get_switch().vlans: 

303 if v_entry.id == vlan: 

304 found_mcast = False 

305 for addr in v_entry.multicast: 

306 if str(addr.address) == mcast_addr: 

307 found_mcast = True 

308 addr.ports.append(comp.name) 

309 if not found_mcast: 

310 new_mcast_group = MulticastGroup(address=mcast_addr, ports=[comp.name]) 

311 v_entry.multicast.append(new_mcast_group)

313 def load_switch_multicast(self, vlans_dict, paths): 

314 for key, value in paths.items(): 

315 for comp in value: 

316 if comp.type == "switch_port": 

317 ip = key.split("/")[0] 

318 self.append_mcast(vlans_dict[key], comp, ip)
