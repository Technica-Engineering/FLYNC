24def validate_vlan_id(value): 

25 """ 

26 Validate a VLAN identifier. 

27 

28 ``None`` is treated as untagged and returned unchanged. 

29 Values in the range 0-4094 are accepted as-is. 

30 The reserved value 4095 is accepted but emits a warning via :func:`warn`. 

31 Anything outside 0-4095 raises a minor validation error.

33 

34 if value is not None: 

35 if value < VLAN_ID_MIN or value > VLAN_ID_MAX:

59def validate_or_remove(label: str, field_type: Any, severity: str = "minor"): 

60 """ 

61 Factory that returns a BeforeValidator for sub-model fields. 

62 

63 Use inside ``Annotated`` to pre-validate a field before Pydantic processes it. 

64 If the raw data fails validation all sub-errors are packed into a single error. 

65 

66 - ``"minor"`` severity: the field is removed and the parent model still loads without it. 

67 The message says "Removing {label}…". 

68 - ``"major"`` severity: the parent model will fail regardless (the field is required). 

69 The message reports the validation failure without implying graceful removal. 

70 

71 The parent object's ``name`` field is included in the error message when available via ``info.data``. 

72 

73 Parameters 

74 ---------- 

75 label : str 

76 Human-readable field label used in the error message. 

77 field_type : Any 

78 Pydantic-compatible type to validate the data against. 

79 severity : str, optional 

80 Error severity — ``"minor"`` (default) or ``"major"``. 

81 

82 Returns 

83 ------- 

84 Callable 

85 A two-argument validator ``(data, info)`` ready for use with ``BeforeValidator``. 

86 """ 

87 

88 err_fn = err_major if severity == "major" else err_minor 

89 

90 def _validator(data, info: ValidationInfo): 

91 """ 

92 Validate ``data`` against ``field_type`` and raise on failure. 

93 

94 Returns ``None`` unchanged. On validation failure, packs all sub-errors into a single ``err_fn`` error whose message includes 

95 the parent object's name (read from ``info.data``) when available. 

96 """ 

97 

98 if data is None: 

99 return None 

100 try: 

101 TypeAdapter(field_type).validate_python(data) 

102 except ValidationError as ve: 

103 parent_name = info.data.get("name") if hasattr(info, "data") and info.data else None 

104 location = f"in {parent_name}" if parent_name else _LOCATION_SYSTEM 

105 sub_errors = "\n".join( 

106 "{loc}: {msg}".format( 

107 loc=".".join(str(x) for x in e.get("loc", ())), 

108 msg=e.get("msg", ""), 

109 ) 

110 for e in ve.errors() 

111 ) 

112 if severity == "major": 

113 raise err_fn( 

114 f"Validation failed for {label} {location}.", 

115 sub_errors=sub_errors, 

116 ) 

117 raise err_fn( 

118 f"1 or more errors found while validating {label}. Removing {label} {location}.", 

119 sub_errors=sub_errors, 

120 ) 

121 return data 

122 

123 return _validator

126def validate_list_items_and_remove(label: str, item_type: Any, severity: str = "minor"): 

127 """ 

128 Validate each item in a list individually, removing only invalid entries. 

129 

130 Unlike :func:`validate_or_remove`, which discards the entire list when any item fails, this validator keeps valid items and removes only those 

131 that fail validation. Per-item errors are forwarded to the ``_validation_warnings`` channel so they appear in the final error report even 

132 though the model continues building with the remaining valid items. 

133 

134 Use inside ``Annotated`` as a ``BeforeValidator``. 

135 

136 Parameters 

137 ---------- 

138 label : str 

139 Human-readable field label used in error messages. 

140 item_type : Any 

141 Pydantic-compatible type for each individual list item. 

142 severity : str, optional 

143 Error severity for removed items — ``"minor"`` (default) or 

144 ``"major"``. 

145 

146 Returns 

147 ------- 

148 Callable 

149 A two-argument validator ``(data, info)`` ready for use with 

150 ``BeforeValidator``. 

151 """ 

152 

153 def _validator(data, info: ValidationInfo): 

154 """ 

155 Validate each item in ``data`` individually, dropping invalid ones. 

156 

157 Non-list values are returned unchanged. For each item that fails validation an error is raised via ``err_fn``; valid items are 

158 collected and returned so the parent model loads with a partial list. The parent object's name or VLAN ID is included in the message when 

159 available via ``info.data``. 

160 """ 

161 

162 if not isinstance(data, list): 

163 return data 

164 location = _resolve_location(info) 

165 field_name = getattr(info, "field_name", None) or label 

166 adapter = TypeAdapter(item_type) 

167 valid_items = [] 

168 for idx, item in enumerate(data): 

169 try: 

170 adapter.validate_python(item) 

171 valid_items.append(item) 

172 except ValidationError as ve: 

173 sub_errors = "\n".join( 

174 "{loc}: {msg}".format( 

175 loc=".".join(str(x) for x in e.get("loc", ())), 

176 msg=e.get("msg", ""), 

177 ) 

178 for e in ve.errors() 

179 ) 

180 accumulated = _validation_warnings.get() 

181 if accumulated is not None: 

182 accumulated.append( 

183 { 

184 "type": severity, 

185 "msg": (f"1 or more errors found while validating {label}. Removing {label} {location}."), 

186 "loc": (field_name, idx), 

187 "input": item, 

188 "ctx": {"sub_errors": sub_errors}, 

189 "url": "", 

190 } 

191 ) 

192 return valid_items 

193 

194 return _validator

378def validate_cbs_idleslopes_fit_portspeed(traffic_classes: list, port_speed: int): 

379 """ 

380 Custom Validator for a list of Traffic Classes to check conformity to MII/MDI speed. 

381 

382 Args: 

383 traffic_classes (list): List of element type `TrafficClass`. 

384 

385 port_speed (int): MII or MDI speed of the port. 

386 

387 Raises: 

388 err_major: The sum of idleslopes of all shapers on one port must be equal or lower than the port speed. 

389 

390 Returns: 

391 list: Return list of traffic classes as received. 

392 """ 

393 

394 if not traffic_classes: 

395 return 

396 if not port_speed: 

397 raise err_major("Cannot validate Traffic Classes! No port speed defined. Make sure to configure MII or MDI.") 

398 

399 sum_idleslopes = 0 

400 

401 for tr_class in traffic_classes: 

402 if tr_class.selection_mechanisms and tr_class.selection_mechanisms.type == "cbs": 

403 sum_idleslopes += tr_class.selection_mechanisms.idleslope 

404 

405 if sum_idleslopes > port_speed * 1000: 

406 raise err_major(("The sum of idleslopes of all shapers on one port" + " cannot be higher than the link speed!")) 

407 return traffic_classes

410def validate_optional_mii_config_compatibility(comp1, comp2, id): 

411 """ 

412 Custom validator for optional MII configuration compatibility between two components. 

413 

414 Args: 

415 comp1 (object): First component that may contain a ``mii_config`` attribute. 

416 

417 comp2 (object): Second component that may contain a ``mii_config`` attribute. 

418 

419 id (Any): Identifier of the connection (used only in error messages). 

420 

421 Raises: 

422 err_major: One component has an MII config while the other does not. 

423 

424 err_major: Both components have an MII config but the *mode* values are identical. The modes must differ. 

425 

426 err_major: Both components have an MII config but the *speed* values are different. 

427 

428 err_major: Both components have an MII config but the *type* values are different. 

429 """ 

430 

431 if not comp1 or not comp2 or not comp1.mii_config or not comp2.mii_config: 

432 return 

433 mii_comp1 = comp1.mii_config 

434 mii_comp2 = comp2.mii_config 

435 # Look for wrong config variants: neither external nor internal PHYs 

436 # used 

437 if (mii_comp1 is None and mii_comp2 is not None) or (mii_comp1 is not None and mii_comp2 is None): 

438 raise err_major( 

439 f"Invalid MII config in connection {id}: " 

440 f"{comp1.name} ↔ {comp2.name} " 

441 f"(MII mismatch for PHY type). Both or None of " 

442 f"the components should have a MII config" 

443 ) 

444 

445 # External PHY is used for this connection 

446 if (mii_comp1 and mii_comp1 is not None) and (mii_comp2 and mii_comp2 is not None): 

447 if mii_comp1.mode == mii_comp2.mode: 

448 raise err_major(f"Incompatible MII Mode: {comp1.name} ({mii_comp1.mode}) ↔ {comp2.name}({mii_comp2.mode})") 

449 if mii_comp1.speed != mii_comp2.speed: 

450 raise err_major(f"Incompatible MII Speed: {comp1.name} ({mii_comp1.speed}) ↔ {comp2.name}({mii_comp2.speed})") 

451 if mii_comp1.type != mii_comp2.type: 

452 raise err_major(f"Incompatible MII Type: {comp1.name} ({mii_comp1.type}) ↔ {comp2.name}({mii_comp2.type})")

642 if not traffic_classes: 

643 return 

644 traffic_class_prios = [] 

645 for traffic_class in traffic_classes: 

646 if traffic_class.priority not in traffic_class_prios: 

647 traffic_class_prios.append(traffic_class.priority) 

648 else: 

649 raise err_minor("Traffic class priority is not unique in controller or switch.")

708# --------------------------------------------------------------------------- 

709# Bit-range placement validators 

710# --------------------------------------------------------------------------- 

711# 

712# These helpers operate on a list of bit ranges expressed as 

713# ``(item_name, start_bit, end_bit_exclusive)`` tuples and are used to 

714# validate placements of :class:`SignalInstance` objects inside a PDU and of 

715# :class:`PDUInstance` objects inside a CAN/LIN frame (when the referenced 

716# PDU's length is known to the caller). 

717 

718BitRange = Tuple[str, int, int] 

719 

720 

721def collect_bit_ranges(items: Iterable[Any], get_range: Callable[[Any], Optional[BitRange]]) -> List[BitRange]: 

722 """ 

723 Build a list of ``(name, start_bit, end_bit_exclusive)`` ranges from ``items``. 

724 

725 ``get_range(item)`` is called for every entry and should return either a 

726 ``(name, start_bit, end_bit_exclusive)`` tuple or ``None`` when the item 

727 is unplaced and should be skipped (e.g. a :class:`SignalInstance` with no 

728 ``bit_position``). 

729 """ 

730 

731 ranges: List[BitRange] = [] 

732 for item in items: 

733 r = get_range(item) 

734 if r is not None: 

735 ranges.append(r) 

736 return ranges 

737 

738 

739def check_bit_ranges_within(context: str, ranges: Iterable[BitRange], max_bits: int) -> None: 

740 """ 

741 Raise :func:`err_minor` when any range extends past ``max_bits``. 

742 

743 ``context`` is a human-readable label of the container (PDU or frame 

744 name) used in the error message. 

745 """ 

746 

747 for item_name, start, end in ranges: 

748 if end > max_bits: 

749 raise err_minor( 

750 "{context}: '{item}' bit range [{start}, {end}) overflows length of {bits} bits", 

751 context=context, 

752 item=item_name, 

753 start=start, 

754 end=end, 

755 bits=max_bits, 

756 ) 

757 

758 

759def check_bit_ranges_no_overlap(context: str, ranges: List[BitRange]) -> None: 

760 """ 

761 Raise :func:`err_minor` when any two ranges in ``ranges`` intersect. 

762 

763 Ranges are half-open ``[start, end)``; two ranges overlap when 

764 ``start_a < end_b and start_b < end_a``. ``context`` is included in the 

765 error message to identify the enclosing PDU or frame. 

766 """ 

767 

768 for i, (name_a, start_a, end_a) in enumerate(ranges): 

769 for j in range(i + 1, len(ranges)): 

770 name_b, start_b, end_b = ranges[j] 

771 if start_a < end_b and start_b < end_a: 

772 raise err_minor( 

773 "{context}: '{a}' [{sa}, {ea}) and '{b}' [{sb}, {eb}) overlap", 

774 context=context, 

775 a=name_a, 

776 sa=start_a, 

777 ea=end_a, 

778 b=name_b, 

779 sb=start_b, 

780 eb=end_b, 

781 )

784def validate_value_input_format(data: dict) -> dict: 

785 """Validating combinations of 'value', 'from_value' and 'to_value'.""" 

786 if not isinstance(data, dict): 

787 return data 

788 

789 has_value = "value" in data 

790 has_from_value = "from_value" in data 

791 has_to_value = "to_value" in data 

792 

793 if not has_value and not has_to_value and not has_from_value: 

794 raise err_major("Field required: Either the field 'value' or the pair of 'from_value' and 'to_value' has to be defined.") 

795 

796 if has_value and has_to_value: 

797 raise err_major( 

798 "Invalid Combination: cannot use both 'value' and 'to_value' — either use 'value' for a single value, " 

799 "or 'from_value' and 'to_value' in a pair." 

800 ) 

801 

802 if has_value and has_from_value: 

803 raise err_major( 

804 "Invalid Combination: cannot use both 'value' and 'from_value' — either use 'value' for a single value, " 

805 "or 'from_value' and 'to_value' in a pair." 

806 ) 

807 

808 if (has_from_value and not has_to_value) or (has_to_value and not has_from_value): 

809 raise err_major( 

810 "Invalid Combination: 'from_value' and 'to_value' must be paired — either use 'value' for a single value, " 

811 "or 'from_value' and 'to_value' in a pair." 

812 ) 

813 

814 return data
