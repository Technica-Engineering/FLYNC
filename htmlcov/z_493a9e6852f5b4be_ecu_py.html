113 def model_post_init(self, context): 

114 """ 

115 Perform post-initialization processing after the model is created. 

116 

117 Following steps are performed: 

118 1. Reference the ECU in child components to allow access to ECU-level information. 

119 

120 2. Bind sockets (defined per ethernet interface) to their corresponding IP addresses in the virtual interfaces, ensuring each socket is 

121 associated with the correct ECU IP. 

122 

123 3. Populate multicast group memberships based on socket configurations and virtual interface settings. 

124 """ 

125 

126 self.__reference_ecu_in_children() 

127 self.__bind_sockets_to_ip() 

128 self.__populate_multicast_tx_groups_from_socket() 

129 self.__populate_multicast_rx_groups_from_interfaces() 

130 self._populate_multicast_tx_groups_from_mac_multicast_endpoints()

132 @model_validator(mode="before") 

133 @classmethod 

134 def skip_on_broken_controllers(cls, data): 

135 """ 

136 Skip ECU validation when controllers or switches failed to load. 

137 

138 When a controller or switch cannot be parsed, the workspace places None in the respective list. 

139 Attempting to validate the ECU in that state produces a cascade of unhelpful errors, 

140 instead report a single major error so that per-component errors remain the focus. 

141 """ 

142 

143 if isinstance(data, dict): 

144 controllers = data.get("controllers") or [] 

145 switches = data.get("switches") or [] 

146 broken_controllers = isinstance(controllers, list) and any(c is None for c in controllers) 

147 broken_switches = isinstance(switches, list) and any(s is None for s in switches) 

148 if broken_controllers or broken_switches: 

149 raise err_major("ECU has invalid components. Check controller and switch errors for details.") 

150 return data

160 @model_validator(mode="after") 

161 def validate_vlans_in_sockets(self): 

162 """ 

163 Validate that the VLAN IDs specified in the socket containers of each ethernet interface are configured in a virtual interface of that same 

164 ethernet interface.""" 

165 

166 for controller in self.controllers: 

167 for eth_iface in controller.ethernet_interfaces: 

168 iface_config = eth_iface.interface_config 

169 vlan_ids_in_sockets = {sc.vlan_id for sc in (eth_iface.sockets or [])} 

170 if not vlan_ids_in_sockets: 

171 continue 

172 vlan_ids_in_interface = {vi.vlanid for vi in iface_config.virtual_interfaces or []} 

173 missing_vlans = vlan_ids_in_sockets - vlan_ids_in_interface 

174 if missing_vlans: 

175 raise err_minor( 

176 "Error in socket configuration:\n" 

177 "The following VLAN IDs are specified in the socket containers " 

178 "but not configured in any virtual interface of ethernet interface " 

179 f"{iface_config.name}: {missing_vlans}." 

180 ) 

181 return self

183 def __bind_sockets_to_ip(self): 

184 """ 

185 Associate each socket with the matching IP address on the same ethernet interface where the socket is defined. 

186 

187 Sockets are scoped to the ethernet interface that owns them, so a socket is only bound to addresses belonging to that interface — not 

188 to identically-addressed interfaces elsewhere in the ECU. 

189 

190 Raises: 

191 err_minor: If a socket's endpoint address does not belong to any virtual interface of the ethernet interface that defines the socket. 

192 """ 

193 

194 for controller in self.controllers: 

195 for eth_iface in controller.ethernet_interfaces: 

196 iface_config = eth_iface.interface_config 

197 iface_ips = set(iface_config.get_all_ips()) 

198 for socket_container in eth_iface.sockets or []: 

199 for socket in socket_container.sockets or []: 

200 if socket.endpoint_type == "multicast": 

201 continue 

202 endpoint_ip = str(socket.endpoint_address) 

203 if endpoint_ip not in iface_ips: 

204 raise err_minor( 

205 f"Error in socket {socket.name}:\n" 

206 f"The IP {endpoint_ip} is not configured in any virtual interface of ethernet " 

207 f"interface {iface_config.name} in ECU {self.name}." 

208 ) 

209 for vi in iface_config.virtual_interfaces: 

210 for ip in vi.addresses: 

211 if ip.address == socket.endpoint_address: 

212 ip.sockets.append(socket) 

213 return self

224 def __populate_multicast_tx_groups_from_socket(self): 

225 """ 

226 Add Multicast TX entries from sockets (defined per ethernet interface) to multicast group memberships. 

227 """ 

228 

229 for controller in self.controllers: 

230 for eth_iface in controller.ethernet_interfaces: 

231 for socket_container in eth_iface.sockets or []: 

232 for socket in socket_container.sockets: 

233 if socket.endpoint_type == "multicast": 

234 for multicast_addr in socket.multicast_tx: 

235 group = MulticastGroupMembership( 

236 group=multicast_addr, 

237 description=socket.name, 

238 mode="tx", 

239 vlan=socket_container.vlan_id, 

240 src_ip=socket.endpoint_address, 

241 ) 

242 interface = self.get_interface_for_ip(str(socket.endpoint_address)) 

243 group._interface = interface 

244 self.multicast_groups.append(group) 

245 return self

247 def __populate_multicast_rx_groups_from_interfaces(self): 

248 """ 

249 Add Multicast RX entries from virtual interfaces to multicast group memberships. 

250 """ 

251 

252 for interface in find_all(self.controllers, ControllerInterface): 

253 for viface in interface.virtual_interfaces: 

254 for multicast_addr in viface.multicast: 

255 group = MulticastGroupMembership( 

256 group=multicast_addr, 

257 description="", 

258 mode="rx", 

259 vlan=viface.vlanid, 

260 src_ip=None, 

261 ) 

262 group._interface = interface 

263 self.multicast_groups.append(group) 

264 for node in interface.compute_nodes: 

265 for viface in node.virtual_interfaces: 

266 for multicast_addr in viface.multicast: 

267 group = MulticastGroupMembership( 

268 group=multicast_addr, 

269 description="", 

270 mode="rx", 

271 vlan=viface.vlanid, 

272 src_ip=None, 

273 ) 

274 group._interface = interface 

275 self.multicast_groups.append(group) 

276 return self

278 def _populate_multicast_tx_groups_from_mac_multicast_endpoints(self): 

279 """ 

280 Add Multicast TX entries from MAC multicast endpoints to multicast group memberships. 

281 """ 

282 

283 if self.mac_multicast_endpoints is not None: 

284 for endpoint_union in self.mac_multicast_endpoints.endpoints: 

285 endpoint = endpoint_union.root 

286 for multicast_addr in endpoint.multicast_tx: 

287 group = MulticastGroupMembership( 

288 group=multicast_addr, 

289 description=endpoint.name, 

290 mode="tx", 

291 vlan=endpoint.vlan_id or None, 

292 src_ip=None, 

293 ) 

294 interface = self.get_interface_for_mac(str(endpoint.mac_address)) 

295 if not interface: 

296 raise err_minor( 

297 f"Error in MAC multicast:\n" 

298 f"endpoints. Could not find an interface for address {endpoint.mac_address}. ECU {self.name}" 

299 ) 

300 group._interface = interface 

301 self.multicast_groups.append(group) 

302 return self

398 def __get_services_of_type(self, service_type: type[_T_Service]) -> list[_T_Service]: 

399 """ 

400 Return all SOME/IP service deployments of a given type. 

401 

402 Iterates over all ethernet interfaces, their socket containers and sockets, collecting deployments whose root matches ``service_type``. 

403 

404 Parameters 

405 ---------- 

406 service_type : type[SOMEIPServiceDeployment] 

407 The concrete deployment type to filter by (e.g. :class:`~flync.model.flync_4_someip.SOMEIPServiceConsumer` or 

408 :class:`~flync.model.flync_4_someip.SOMEIPServiceProvider`). 

409 

410 Returns 

411 ------- 

412 list[SOMEIPServiceDeployment] 

413 All matching service deployment instances found across the ECU's socket containers. 

414 """ 

415 

416 service_instances = [] 

417 for controller in self.controllers: 

418 for eth_iface in controller.ethernet_interfaces or []: 

419 for ecu_sockets in eth_iface.sockets or []: 

420 for socket in ecu_sockets.sockets or []: 

421 for deployment in socket.deployments or []: 

422 if isinstance(deployment.root, service_type): 

423 someip_deployment = deployment.root 

424 service_instances.append(someip_deployment) 

425 return service_instances
